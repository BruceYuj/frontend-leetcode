# 简介

  • 297. 二叉树的序列化与反序列化

序列化和反序列化在跨系统、跨平台是很常用的功能。 题目要求的基础数据是二叉树。而表示一棵二叉树，深度优先和宽度优先是很好的手段之一。

# 解法一 - BFS

/**
 * Definition for a binary tree node.
 * function TreeNode(val) {
 *     this.val = val;
 *     this.left = this.right = null;
 * }
 */

/**
 * Encodes a tree to a single string.
 *
 * @param {TreeNode} root
 * @return {string}
 */
var serialize = function(root) {
    let str = '';
    if(!root) return str;

    let queue = [root];

    while(queue.length) {
        let top = queue.shift();
        if(!top) {
            str += 'null,';
            continue;
        }
        str += top.val + ',';

        queue.push(top.left, top.right);
        
    }

    return str.slice(0, -1);
};

/**
 * Decodes your encoded data to tree.
 *
 * @param {string} data
 * @return {TreeNode}
 */
var deserialize = function(data) {
    if(!data.length) return null;

    let arr = data.split(',');
    let root = new TreeNode(arr[0]);
    let queue = [root];
    let count = 1;

    while(count < arr.length) {
        let top = queue.shift();
        if(arr[count] !== 'null') {
            top.left = new TreeNode(arr[count]);
            queue.push(top.left);
        }
        if(arr[count+1] !== 'null') {
            top.right = new TreeNode(arr[count+1]);
            queue.push(top.right);
        }

        count += 2;
    }

    return root;
};

/**
 * Your functions will be called as such:
 * deserialize(serialize(root));
 */

复杂度分析:

  • 时间复杂度：序列化和反序列化都是 , 是二叉树结点数量
  • 空间复杂度：反序列化需要额外的 ，这里不考虑被序列化的结果

# 解法二 - DFS

/**
 * Definition for a binary tree node.
 * function TreeNode(val) {
 *     this.val = val;
 *     this.left = this.right = null;
 * }
 */

/**
 * Encodes a tree to a single string.
 *
 * @param {TreeNode} root
 * @return {string}
 */
var serialize = function(root) {
    let str = '';

    dfs(root);
    return str.slice(0, -1);

    function dfs(root) {
        if(!root) {
            str += 'null,';
            return; 
        }
        str += root.val + ',';
        dfs(root.left);
        dfs(root.right);
    }  
};

/**
 * Decodes your encoded data to tree.
 *
 * @param {string} data
 * @return {TreeNode}
 */
var deserialize = function(data) {
    let arr = data.split(',');
    return  dfs(arr);

    function dfs(arr) {
        if(arr[0] === 'null') {
            arr.shift();
            return null;
        }

        let root = new TreeNode(arr[0]);
        arr.shift();
        root.left = dfs(arr);
        root.right = dfs(arr);

        return root;
    }
};

/**
 * Your functions will be called as such:
 * deserialize(serialize(root));
 */

复杂度分析:

  • 时间复杂度：序列化和反序列化都是 , 是二叉树结点数量
  • 空间复杂度：序列化和反序列化的递归深度最多是 , 反序列化需要额外的
