## 快慢指针
先找到中点，然后翻转前半部分，然后比较回文。

```javascript
/**
 * Definition for singly-linked list.
 * function ListNode(val) {
 *     this.val = val;
 *     this.next = null;
 * }
 */
/**
 * @param {ListNode} head
 * @return {boolean}
 */
var isPalindrome = function(head) {
    if(!head || !head.next) return true;

    let slow = head;
    let fast = head.next;

    while(fast && fast.next) {
        fast = fast.next.next;
        slow = slow.next;
    }

    let right = slow.next;

    let prev = null;
    let curr = head;
    while(curr !== right) {
        let tmp = curr.next;
        curr.next = prev;
        prev =curr;
        curr = tmp;
    }

    if(!fast) slow =slow.next;

    while(slow) {
        if(slow.val === right.val) {
            slow = slow.next;
            right = right.next;
        }else {
            return false;
        }
    } 

    return true;

};
```
**复杂度分析**:
- 时间复杂度：$O(N)$
- 空间复杂度：$O(1)$
